    import numpy as np
    import pooch
    import pyvista as pv
    import xarray as xr

    url = "https://raw.githubusercontent.com/pyvista/vtk-data/master/Data/lsm_4x5.nc"
    file_path = pooch.retrieve(url=url, known_hash=None)

    lsm = xr.open_dataarray(file_path)

    xx, yy, zz = np.meshgrid(
